-- De modulus: m = p * q

modulus_m :: Double -> Double  -> Double
modulus_m p q = p * q

-- Eulers totient functie: m0 = phi(m) = (p - 1) * (q - 1)
euler_totient :: Double -> Double  -> Double
euler_totient p q = (p - 1) * (q - 1)
